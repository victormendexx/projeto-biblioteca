Table Livro {
  ID int [primary key]
  Titulo varchar
  Ano_de_Publicacao varchar
  Idioma varchar
  Imagem varchar
  Sinopse text
  AutorID int [ref: > Autor.ID]
  GeneroID int [ref: > Genero.ID]
  EditoraID int [ref: > Editora.ID]
}

Table Autor {
  ID int [primary key]
  Nome_do_Autor varchar
}

Table Genero {
  ID int [primary key]
  Nome_do_Genero varchar
}

Table Editora {
  ID int [primary key]
  Nome_da_Editora varchar
}
